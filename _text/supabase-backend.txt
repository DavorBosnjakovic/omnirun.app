CREATE OR REPLACE FUNCTION "public"."check_project_limit"() RETURNS json
    LANGUAGE "plpgsql" SECURITY DEFINER
    AS $$
DECLARE
  user_plan TEXT;
  project_count INTEGER;
  project_limit INTEGER;
BEGIN
  -- Get user's current plan
  SELECT plan INTO user_plan
  FROM profiles
  WHERE id = auth.uid();

  -- Count ALL registered projects (active + inactive)
  SELECT COUNT(*) INTO project_count
  FROM user_projects
  WHERE user_id = auth.uid();

  -- Get limit for their plan
  project_limit := get_project_limit(COALESCE(user_plan, 'starter'));

  RETURN json_build_object(
    'allowed', project_count < project_limit,
    'current', project_count,
    'limit', project_limit,
    'plan', COALESCE(user_plan, 'starter')
  );
END;
$$;


ALTER FUNCTION "public"."check_project_limit"() OWNER TO "postgres";


CREATE OR REPLACE FUNCTION "public"."check_seat_limit"() RETURNS "trigger"
    LANGUAGE "plpgsql"
    AS $$
DECLARE
  current_count INTEGER;
  max_allowed INTEGER;
BEGIN
  SELECT COUNT(*) INTO current_count
  FROM team_members WHERE team_id = NEW.team_id;

  SELECT max_seats INTO max_allowed
  FROM teams WHERE id = NEW.team_id;

  IF current_count >= max_allowed THEN
    RAISE EXCEPTION 'Team has reached maximum seat limit (%)', max_allowed;
  END IF;

  RETURN NEW;
END;
$$;


ALTER FUNCTION "public"."check_seat_limit"() OWNER TO "postgres";

CREATE OR REPLACE FUNCTION "public"."handle_new_user"() RETURNS "trigger"
    LANGUAGE "plpgsql" SECURITY DEFINER
    SET "search_path" TO 'public'
    AS $$
DECLARE
  detected_provider TEXT;
BEGIN
  -- Detect auth provider from Supabase auth metadata
  detected_provider := COALESCE(
    NEW.raw_app_meta_data->>'provider',
    'email'
  );

  INSERT INTO profiles (id, email, display_name, avatar_url, auth_provider, subscription_status)
  VALUES (
    NEW.id,
    NEW.email,
    COALESCE(
      NEW.raw_user_meta_data->>'full_name',
      NEW.raw_user_meta_data->>'name',
      split_part(NEW.email, '@', 1)
    ),
    NEW.raw_user_meta_data->>'avatar_url',
    detected_provider,
    'incomplete'
  );
  RETURN NEW;
END;
$$;


ALTER FUNCTION "public"."handle_new_user"() OWNER TO "postgres";

CREATE OR REPLACE FUNCTION "public"."register_project"("p_path_hash" "text", "p_project_name" "text", "p_machine_id" "text") RETURNS json
    LANGUAGE "plpgsql" SECURITY DEFINER
    AS $$
DECLARE
  user_plan TEXT;
  project_count INTEGER;
  project_limit INTEGER;
  existing_project user_projects%ROWTYPE;
  new_project user_projects%ROWTYPE;
BEGIN
  -- Check if this exact project already exists (re-adding a removed folder)
  SELECT * INTO existing_project
  FROM user_projects
  WHERE user_id = auth.uid() AND path_hash = p_path_hash;

  IF FOUND THEN
    -- Re-activate existing project ÔÇö doesn't count as new
    UPDATE user_projects
    SET status = 'active',
        project_name = p_project_name,
        last_opened_at = now(),
        removed_at = NULL
    WHERE id = existing_project.id
    RETURNING * INTO new_project;

    RETURN json_build_object(
      'success', true,
      'reactivated', true,
      'project_id', new_project.id
    );
  END IF;

  -- New project ÔÇö check limit
  SELECT plan INTO user_plan FROM profiles WHERE id = auth.uid();
  SELECT COUNT(*) INTO project_count FROM user_projects WHERE user_id = auth.uid();
  project_limit := get_project_limit(COALESCE(user_plan, 'starter'));

  IF project_count >= project_limit THEN
    RETURN json_build_object(
      'success', false,
      'error', 'project_limit_reached',
      'current', project_count,
      'limit', project_limit,
      'plan', COALESCE(user_plan, 'starter')
    );
  END IF;

  -- Insert new project
  INSERT INTO user_projects (user_id, path_hash, project_name, machine_id)
  VALUES (auth.uid(), p_path_hash, p_project_name, p_machine_id)
  RETURNING * INTO new_project;

  RETURN json_build_object(
    'success', true,
    'reactivated', false,
    'project_id', new_project.id
  );
END;
$$;


ALTER FUNCTION "public"."register_project"("p_path_hash" "text", "p_project_name" "text", "p_machine_id" "text") OWNER TO "postgres";

CREATE OR REPLACE VIEW "public"."admin_cohort_retention" AS
 SELECT "to_char"("p"."created_at", 'YYYY-MM'::"text") AS "cohort_month",
    "count"(DISTINCT "p"."id") AS "cohort_size",
    "count"(DISTINCT
        CASE
            WHEN (("s"."started_at" >= "p"."created_at") AND ("s"."started_at" < ("p"."created_at" + '30 days'::interval))) THEN "p"."id"
            ELSE NULL::"uuid"
        END) AS "active_month_0",
    "count"(DISTINCT
        CASE
            WHEN (("s"."started_at" >= ("p"."created_at" + '30 days'::interval)) AND ("s"."started_at" < ("p"."created_at" + '60 days'::interval))) THEN "p"."id"
            ELSE NULL::"uuid"
        END) AS "active_month_1",
    "count"(DISTINCT
        CASE
            WHEN (("s"."started_at" >= ("p"."created_at" + '60 days'::interval)) AND ("s"."started_at" < ("p"."created_at" + '90 days'::interval))) THEN "p"."id"
            ELSE NULL::"uuid"
        END) AS "active_month_2",
    "count"(DISTINCT
        CASE
            WHEN (("s"."started_at" >= ("p"."created_at" + '90 days'::interval)) AND ("s"."started_at" < ("p"."created_at" + '120 days'::interval))) THEN "p"."id"
            ELSE NULL::"uuid"
        END) AS "active_month_3",
    "count"(DISTINCT
        CASE
            WHEN (("s"."started_at" >= ("p"."created_at" + '120 days'::interval)) AND ("s"."started_at" < ("p"."created_at" + '150 days'::interval))) THEN "p"."id"
            ELSE NULL::"uuid"
        END) AS "active_month_4",
    "count"(DISTINCT
        CASE
            WHEN (("s"."started_at" >= ("p"."created_at" + '150 days'::interval)) AND ("s"."started_at" < ("p"."created_at" + '180 days'::interval))) THEN "p"."id"
            ELSE NULL::"uuid"
        END) AS "active_month_5"
   FROM ("public"."profiles" "p"
     LEFT JOIN "public"."app_sessions" "s" ON (("s"."user_id" = "p"."id")))
  GROUP BY ("to_char"("p"."created_at", 'YYYY-MM'::"text"))
  ORDER BY ("to_char"("p"."created_at", 'YYYY-MM'::"text")) DESC;


ALTER VIEW "public"."admin_cohort_retention" OWNER TO "postgres";

CREATE OR REPLACE VIEW "public"."admin_feature_adoption" AS
 SELECT "feature",
    "count"(*) AS "total_uses",
    "count"(DISTINCT "user_id") AS "unique_users",
    "count"(*) FILTER (WHERE ("occurred_at" >= ("now"() - '7 days'::interval))) AS "uses_7d",
    "count"(*) FILTER (WHERE ("occurred_at" >= ("now"() - '30 days'::interval))) AS "uses_30d",
    "count"(DISTINCT "user_id") FILTER (WHERE ("occurred_at" >= ("now"() - '30 days'::interval))) AS "unique_users_30d",
    "min"("occurred_at") AS "first_used",
    "max"("occurred_at") AS "last_used"
   FROM "public"."feature_events"
  GROUP BY "feature"
  ORDER BY ("count"(*) FILTER (WHERE ("occurred_at" >= ("now"() - '30 days'::interval)))) DESC;
